34  // conversion between two fixed_point types where rounding *is* an issue
35  template<
36  typename ResultRep, int ResultExponent,
37  typename InputRep, int InputExponent,
38  int Radix>
39  struct convert<
40  _impl::nearest_rounding_tag,
41  fixed_point<ResultRep, ResultExponent, Radix>,
42  fixed_point<InputRep, InputExponent, Radix>,
43  _impl::enable_if_t<(ResultExponent>InputExponent)>> {
44  private:
45  using _result = fixed_point<ResultRep, ResultExponent, Radix>;
46  using _input = fixed_point<InputRep, InputExponent, Radix>;
47 
48  static constexpr _input half()
49  {
50  return static_cast<_input>(_impl::from_rep<_result>(1))/2;
51  }
52 
53  public:
54  constexpr _result operator()(_input const& from) const
55  {
56  // TODO: unsigned specialization
57  return static_cast<_result>(from+((from>=0) ? half() : -half()));
58  }
59  };

61  // conversion from float to fixed_point
62  template<
63  typename ResultRep, int ResultExponent, int ResultRadix,
64  typename Input>
65  struct convert<
66  _impl::nearest_rounding_tag,
67  fixed_point<ResultRep, ResultExponent, ResultRadix>,
68  Input,
69  _impl::enable_if_t<std::is_floating_point<Input>::value>> {
70  private:
71  using _result = fixed_point<ResultRep, ResultExponent, ResultRadix>;
72 
73  static constexpr Input half()
74  {
75  return power<Input, ResultExponent-1, ResultRadix>();
76  }
77 
78  public:
79  constexpr _result operator()(Input const& from) const
80  {
81  // TODO: unsigned specialization

83  }
