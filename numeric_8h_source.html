301  // cnl::used_digits
302 
303  namespace _numeric_impl {
304  template<int Radix, typename Integer>
305  constexpr int used_digits_positive(Integer const& value)
306  {
307  static_assert(cnl::numeric_limits<Integer>::is_integer,
308  "Integer parameter of used_digits_positive() must be a fundamental integer.");
309 
310  return (value>0) ? 1+used_digits_positive<Radix>(value/Radix) : 0;
311  }
312 
313  template<int Radix, bool IsSigned>
314  struct used_digits {
315  template<class Integer>
316  constexpr int operator()(Integer const& value) const
317  {
318  return value ? used_digits_positive<Radix>(value) : 0;
319  }
320  };
321 
322  template<int Radix>
323  struct used_digits<Radix, true> {
324  template<class Integer>
325  constexpr int operator()(Integer const& value) const
326  {
327  static_assert(cnl::numeric_limits<Integer>::is_integer,
328  "Integer parameter of used_digits()() must be a fundamental integer.");
329 
330  // Most negative number is not exploited;
331  // thus negating the result or subtracting it from something else
332  // will less likely result in overflow.
333  return (value>0)
334  ? used_digits_positive<Radix>(value)
335  : (value==0)
336  ? 0
337  : used_digits()(Integer(-1)-value);
338  }
339  };
340  }
341 
342  template<int Radix = 2, typename Integer=void>
343  constexpr int used_digits(Integer const& value)
344  {
345  return _impl::for_rep<int>(_numeric_impl::used_digits<Radix, is_signed<Integer>::value>(), value);
346  }
