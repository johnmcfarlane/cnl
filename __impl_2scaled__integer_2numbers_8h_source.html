28  template<typename Rep, int Exponent>
29  [[nodiscard]] constexpr auto pi(int const max_iterations)
30  {
31  using fp = scaled_integer<Rep, power<Exponent>>;
32  constexpr auto four = scaled_integer<Rep, power<3 - digits_v<Rep>>>{4.};
33 
34  auto previous = fp{3.};
35  for (auto n = 2; n != (max_iterations << 1); n += 4) {
36  auto const addend = four / ((n + 0L) * (n + 1L) * (n + 2L));
37  auto const subtrahend = four / ((n + 2L) * (n + 3L) * (n + 4L));
38  auto next = fp{previous + addend - subtrahend};
39  if (next == previous) {
40  return next;
41  }
42  previous = next;
43  }
44  return previous;
45  }

53  template<typename Rep, int Exponent>
54  [[nodiscard]] constexpr auto e(int const max_iterations)
55  {
56  using fp = scaled_integer<Rep, power<Exponent>>;
57  constexpr auto one = scaled_integer<Rep, power<2 - digits_v<Rep>>>{1.};
58 
59  auto previous = fp{2.};
60  auto factor = 2;
61  for (auto n = 2; n != max_iterations; ++n, factor *= n) {
62  auto const addend = one / factor;
63  auto next = fp{previous + addend};
64  if (next == previous) {
65  return next;
66  }
67  previous = next;
68  }
69  return previous;
70  }

78  // Given two alternative ways to generate a constant value:
79  // tries to choose the best and returns the result.
80  template<typename Float, typename Rep, int Exponent>
81  [[nodiscard]] constexpr auto constant_with_fallback(
82  // NOLINTNEXTLINE(misc-unused-parameters)
83  Float constant, scaled_integer<Rep, power<Exponent>> (*procedure)())
84  {
85  using fp = scaled_integer<Rep, power<Exponent>>;
86 
87  auto const required_integer_digits = used_digits(static_cast<int>(constant));
88 
89  constexpr auto fixed_fractional_digits = fractional_digits<fp>;
90  // expect: integer_digits<fp> >= required_integer_digits
91 
92  constexpr auto float_digits = std::numeric_limits<Float>::digits;
93  auto const float_fractional_digits = float_digits - required_integer_digits;
94 
95  return (float_fractional_digits >= fixed_fractional_digits) ? fp{constant}
96  : procedure();
97  }
98  }
