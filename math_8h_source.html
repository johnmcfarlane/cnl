89  template<class Rep, int Exponent>
90  constexpr inline fixed_point<Rep, Exponent> evaluate_polynomial(
91  fixed_point<Rep, Exponent> xf) {
92  using fp = fixed_point<Rep, Exponent>;
93 
94  //Use a polynomial min-max approximation to generate the exponential of
95  //the fraction part. Note that the constant 1 of the polynomial is added later,
96  //this gives us one more bit of precision here for free
97  using coeffs = poly_coeffs<fp>;
98  return fp{
99  safe_multiply(xf, (coeffs::a1+fp{
100  safe_multiply(xf, (coeffs::a2+fp{
101  safe_multiply(xf, (coeffs::a3+fp{
102  safe_multiply(xf, (coeffs::a4+fp{
103  safe_multiply(xf, (coeffs::a5+fp{
104  safe_multiply(xf, (coeffs::a6+fp{
105  safe_multiply(coeffs::a7, xf)
106  }))
107  }))
108  }))
109  }))
110  }))
111  }))
112  };
113  }

121  // Cannot construct from 0, since that would be a shift by more than width of type!

125  //for a positive exponent, some work needs to be done
126  template<class Rep, int Exponent>
127  constexpr inline auto exp2m1_0to1(fixed_point<Rep, Exponent> x)
128  -> _impl::enable_if_t<(Exponent<0), make_largest_ufraction<fixed_point<Rep, Exponent>>> {
129  //Build the type with the same number of bits, all fraction,
130  //and unsigned. That should be enough to exactly hold enough bits
131  //to guarantee bit-accurate results
132  using im = make_largest_ufraction<fixed_point<Rep, Exponent>>;
133  //The intermediate value type
134 
135  return evaluate_polynomial(im{fixed_point<typename im::rep, Exponent>{x}});
136  }

150  template<class Intermediate, typename Rep, int Exponent>
151  constexpr typename Intermediate::rep
152  exp2(fixed_point<Rep, Exponent> const& x, Rep const& floored) {
153  return floored <= Exponent
154  ? typename Intermediate::rep{1}//return immediately if the shift would result in all bits being shifted out
155  //Do the shifts manually. Once the branch with shift operators is merged, could use those
156  : (_impl::to_rep(exp2m1_0to1<Rep, Exponent>(fractional(x, floored)))//Calculate the exponent of the fraction part
157  >> (-Intermediate::exponent + Exponent - floored))//shift it to the right place
158  + (Rep { 1 } << (floored - Exponent)); //The constant term must be one, to make integer powers correct
159  }
160  }

171  template<class Rep, int Exponent>
172  constexpr fixed_point<Rep, Exponent> exp2(fixed_point<Rep, Exponent> x) {
173  using out_type = fixed_point<Rep, Exponent>;
174  // The input type
175  using im = _impl::fp::make_largest_ufraction<out_type>;
176 
177  //Calculate the final result by shifting the fraction part around.
178  //Remember to add the 1 which is left out to get 1 bit more resolution
179  return _impl::from_rep<out_type>(_impl::fp::exp2<im>(x, static_cast<Rep>(floor(x))));
180  }
