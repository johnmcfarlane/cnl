61  template<typename ParseDigit>
62  constexpr intmax parse(char const* s, int base, ParseDigit parse_digit, intmax value = 0)
63  {
64  return (*s) ? parse(s+1, base, parse_digit, parse_digit(*s)+value*base) : value;
65  }

70  constexpr int parse_bin_char(char c) {
71  return (c == '0') ? 0 : (c == '1') ? 1 : int{};
72  }
73 
74  constexpr int parse_dec_char(char c) {
75  return (c >= '0' && c <= '9') ? c - '0' : int{};
76  }
77 
78  constexpr int parse_oct_char(char c) {
79  return (c >= '0' && c <= '7') ? c - '0' : int{};
80  }
81 
82  constexpr int parse_hex_char(char c) {
83  return (c >= '0' && c <= '9')
84  ? c - '0'
85  : (c >= 'a' && c <= 'z')
86  ? c + 10 - 'a'
87  : (c >= 'A' && c <= 'Z')
88  ? c + 10 - 'A'
89  : int{};
90  }
91 
92  template<int NumChars>
93  constexpr intmax parse(const char (& s)[NumChars])
94  {
95  return (s[0]!='0')
96  ? parse(s, 10, parse_dec_char)
97  : (s[1]=='x' || s[1]=='X')
98  ? parse(s+2, 16, parse_hex_char)
99  : (s[1]=='b' || s[1]=='B')
100  ? parse(s+2, 2, parse_bin_char)
101  : parse(s+1, 8, parse_oct_char);
102  }
103 
104  template<char... Chars>
105  constexpr intmax parse() {
106  return parse<sizeof...(Chars) + 1>({Chars...,'\0'});
107  }
108  }
